\section{Project Background}

\begin{frame}{Project Background}
    \small
    \begin{block}{Background}
        \begin{itemize}
            \item Test design often depends on manual extraction of coverage items, risks, and strategies from requirements, which is costly and inconsistent.
            \item FitnessAI contains pose analysis, state-machine counting, record filtering, and statistical aggregation, making it a suitable validation target for a test design tool.
            \item Therefore, we implemented AutoTestDesign to connect requirement analysis, risk assessment, test design, and interactive review into one end-to-end workflow.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Project Objectives}
    \small
    \begin{block}{Objectives}
        \begin{itemize}
            \item Build a runnable AI-driven test design tool with requirement structuring, QRA, black-box and white-box design, and artifact export.
            \item Apply the tool to FitnessAI instead of keeping it as a standalone demo.
            \item Validate the output through automated execution, Allure, and JaCoCo.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Overall Solution}
    \small
    \begin{block}{Architecture and Validation Path}
        \begin{itemize}
            \item Tool: AutoTestDesign (Vue 3 + Express + FastAPI + PostgreSQL)
            \item Target: FitnessAI intelligent fitness assistant system
            \item Validation path: QRA risk analysis -> test plan -> detailed test design -> automated execution and coverage evidence
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Project Scope and Outputs}
    \begin{columns}[T]
        \column{0.42\textwidth}
        \footnotesize
        \begin{itemize}
            \item Main endpoints:
            \item \texttt{/api/analytics/pose}
            \item \texttt{/api/user/\{id\}/records}
            \item \texttt{/api/user/\{id\}/dashboard}
            \item Extra scope: history filters, \texttt{/api/exercises}, and reset
            \item Tool output: 10 requirements, 10 risks, 37 coverage items, 81 test cases, 6 methods
